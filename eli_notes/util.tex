\newcommand{\sga}[6]
{
    % analyzer body
    \draw[rounded corners] (#3-1, #4-0.5) rectangle (#3+1, #4+0.5) {};
    % analyzer labels
    \draw (#3, #4+0.75) node[font=\small] {#2};
    \draw (#3, #4) node {#1};
    % output high
    \draw (#3+0.75, #4+0.2) node {$+$};
    \draw [thick, ->, >=stealth'](#3+1,#4+0.2) -- (#3+2,#4+0.2);
    \draw (#3+2.4, #4+0.2) node {$#5$};
    % output low
    \draw (#3+0.75, #4-0.2) node {$-$};
    \draw [thick, ->, >=stealth'](#3+1,#4-0.2) -- (#3+2,#4-0.2);
    \draw (#3+2.4, #4-0.2) node {$#6$};
}
